@app.route("/sa/config") @login_required @require_roles("superadmin") @no_cache def sa_config(): ed = empresas_data() pd_ = proyectos_data() ud = usuarios_data() empresas = ed.get("empresas", []) proyectos = pd_.get("proyectos", []) usuarios = ud.get("usuarios", []) # Mapas por empresa (sirven igual aunque sea panel único) proys_por_empresa = {} for p in proyectos: proys_por_empresa.setdefault(p.get("empresa_id"), []).append(p) users_por_empresa = {} for u in usuarios: users_por_empresa.setdefault(u.get("empresa_id"), []).append(u) # Ordenar empresas empresas_sorted = sorted(empresas, key=lambda x: (str(x.get("nombre", "")).lower(), x.get("id", 0))) # empresa seleccionada (por querystring) o primera disponible empresa_id_raw = request.args.get("empresa_id") empresa_sel = None if empresa_id_raw: try: eid = int(empresa_id_raw) empresa_sel = next((e for e in empresas_sorted if e.get("id") == eid), None) except Exception: empresa_sel = None if not empresa_sel and empresas_sorted: empresa_sel = empresas_sorted[0] # Listados filtrados de la empresa seleccionada (si existe) proyectos_sel = proys_por_empresa.get(empresa_sel.get("id"), []) if empresa_sel else [] usuarios_sel = users_por_empresa.get(empresa_sel.get("id"), []) if empresa_sel else [] # Ordenar listados seleccionados proyectos_sel = sorted(proyectos_sel, key=lambda x: (str(x.get("nombre", "")).lower(), x.get("id", 0))) usuarios_sel = sorted(usuarios_sel, key=lambda x: (str(x.get("correo", "")).lower(), x.get("id", 0))) return render_template( "sa_config.html", empresas=empresas_sorted, empresa_sel=empresa_sel, proyectos_sel=proyectos_sel, usuarios_sel=usuarios_sel, data_dir=DATA_DIR, # Mantengo también estos por si el template viejo los usa en algún bloque proys_por_empresa=proys_por_empresa, users_por_empresa=users_por_empresa, proyectos=proyectos, usuarios=usuarios )
